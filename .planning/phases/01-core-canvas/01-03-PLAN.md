<objective>
Build the CanvasWrapper component that bridges Excalidraw with the Zustand stores, and integrate it into the App shell.

Purpose: CanvasWrapper is the critical integration point — it mounts the Excalidraw React component, wires its onChange callback to CanvasStore, disables native undo (D-10), implements the local-state-for-drag pattern (D-23), and triggers HistoryStore snapshots. It also stores the Excalidraw imperative API reference in CanvasStore via setExcalidrawAPI, fulfilling the cross-plan contract required by Plan 04's layer panel operations.

**Cross-plan contract:** CanvasWrapper calls `useCanvasStore.getState().setExcalidrawAPI(api)` in the `excalidrawAPI` callback when Excalidraw mounts. This makes the API reference available to Plan 04's LayerPanel layer operations (`_updateExcalidrawScene` calls in move/reorder actions).

Output: A working canvas in the browser where users can draw shapes, pan, zoom, and undo/redo their actions.
</objective>

    Create `apps/web/src/components/CanvasWrapper.tsx` — the bridge component.

    Per D-23 (local-state-for-drag): The component uses local state during drag operations and only commits to CanvasStore on pointer release. This is implemented via:
    - A `isPointerDownRef` (useRef) tracking pointer state
    - Excalidraw's `onChange` callback checks `isPointerDownRef.current` — if true, it does NOT commit to CanvasStore (local state only)
    - A `onPointerUp` handler commits the final state to CanvasStore and triggers history snapshot

    Per D-10 (disable native undo): Two mechanisms:
    1. `captureUpdate: 'never'` in the `initialData.appState` (or in each `updateScene()` call)
    2. `UIOptions={{ canvasActions: { undo: false, redo: false } }}` to hide the undo/redo buttons

    **Cross-plan contract: setExcalidrawAPI MUST be called here.**
    Within the `excalidrawAPI` callback prop, call `useCanvasStore.getState().setExcalidrawAPI(api)`.
    This stores the API reference in CanvasStore so that Plan 04's `_updateExcalidrawScene()` can call `api.updateScene()`.

    Implementation details:

    ```typescript
    import { useCallback, useEffect, useRef, useState } from 'react'
    import { Excalidraw } from '@ac-canvas/excalidraw'  // Map from actual exports
    import { useShallow } from 'zustand/react/shallow'
    import { useCanvasStore } from '../stores/canvasStore'
    import { useHistoryStore } from '../stores/historyStore'

    export function CanvasWrapper() {
      const excalidrawRef = useRef<any>(null)
      const isPointerDownRef = useRef(false)
      const [isReady, setIsReady] = useState(false)

      // Fine-grained selectors per D-24 — only subscribe to what we need
      const elements = useCanvasStore(
        useShallow((s) => Object.values(s.elements))
      )
      const viewport = useCanvasStore(
        useShallow((s) => ({ ...s.viewport }))
      )
      const isDragging = useCanvasStore((s) => s.isDragging)

      // Wire Excalidraw onChange to CanvasStore
      const handleChange = useCallback(
        (excalidrawElements: readonly any[], appState: any, _files: any) => {
          // D-23: During drag, don't commit to global store
          if (isPointerDownRef.current) return

          // Commit to CanvasStore
          useCanvasStore.getState().setElements(excalidrawElements)
          useCanvasStore.getState().setViewport({
            x: appState.scrollX ?? 0,
            y: appState.scrollY ?? 0,
            zoom: appState.zoom?.value ?? 1,
          })

          // D-09: Capture history snapshot (HistoryStore checks isPaused internally)
          useHistoryStore.getState().captureSnapshot()
        },
        []
      )

      // Pointer event handlers for D-23 local-state-for-drag
      const handlePointerDown = useCallback(() => {
        isPointerDownRef.current = true
        useCanvasStore.getState().setIsDragging(true)
        useHistoryStore.getState().setPaused(true)
      }, [])

      const handlePointerUp = useCallback(() => {
        isPointerDownRef.current = false
        useCanvasStore.getState().setIsDragging(false)
        useHistoryStore.getState().setPaused(false)

        // Commit final state after drag
        if (excalidrawRef.current) {
          const sceneElements = excalidrawRef.current.getSceneElements()
          const appState = excalidrawRef.current.getAppState()
          if (sceneElements) {
            useCanvasStore.getState().setElements(sceneElements)
          }
          if (appState) {
            useCanvasStore.getState().setViewport({
              x: appState.scrollX ?? 0,
              y: appState.scrollY ?? 0,
              zoom: appState.zoom?.value ?? 1,
            })
          }
        }

        // Capture one snapshot after drag completes
        useHistoryStore.getState().captureSnapshot()
      }, [])

      // Keyboard shortcut: Ctrl+Z / Cmd+Z for undo, Ctrl+Shift+Z / Cmd+Shift+Z for redo
      useEffect(() => {
        const handleKeyDown = (e: KeyboardEvent) => {
          const isMod = e.metaKey || e.ctrlKey
          if (isMod && e.key === 'z' && !e.shiftKey) {
            e.preventDefault()
            useHistoryStore.getState().undo()
          }
          if (isMod && e.key === 'z' && e.shiftKey) {
            e.preventDefault()
            useHistoryStore.getState().redo()
          }
          if (isMod && e.key === 'Z') {
            e.preventDefault()
            useHistoryStore.getState().redo()
          }
        }
        window.addEventListener('keydown', handleKeyDown)
        return () => window.removeEventListener('keydown', handleKeyDown)
      }, [])

      // Mount UI
      return (
        <div
          className="w-full h-full relative"
          onPointerDown={handlePointerDown}
          onPointerUp={handlePointerUp}
          style={{ touchAction: 'none' }}
        >
          {!isReady && (
            <div className="absolute inset-0 flex items-center justify-center text-gray-400 z-10 bg-white">
              Loading canvas...
            </div>
          )}
          {/*
            D-10: Disable native undo via:
            1. captureUpdate: 'never' in initialData.appState
            2. UIOptions to hide undo/redo buttons
            D-06: Keep full Excalidraw toolbar/menus intact

            CROSS-PLAN CONTRACT: The excalidrawAPI callback stores the API reference
            in CanvasStore so Plan 04's layer operations can use updateScene().
          */}
          <Excalidraw
            excalidrawAPI={(api: any) => {
              excalidrawRef.current = api
              // Store API in CanvasStore for Plan 04 layer operations
              useCanvasStore.getState().setExcalidrawAPI(api)
              setIsReady(true)
            }}
            onChange={handleChange}
            initialData={{
              elements: elements.length > 0 ? elements : undefined,
              appState: {
                scrollX: viewport.x,
                scrollY: viewport.y,
                zoom: { value: viewport.zoom },
              },
            }}
            UIOptions={{
              canvasActions: {
                undo: false,
                redo: false,
              },
            }}
          />
        </div>
      )
    }
    ```
