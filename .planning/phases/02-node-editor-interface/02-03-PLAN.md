    Create `packages/node-editor/src/ConnectionValidator.ts` with:

    ```typescript
    export interface ConnectionValidationResult {
      valid: boolean
      reason?: string
    }

    export function validateConnection(
      source: string,
      target: string,
      sourceHandle: string | null,
      targetHandle: string | null,
      existingEdges: { source: string; target: string; sourceHandle?: string; targetHandle?: string }[]
    ): ConnectionValidationResult {
      // Step 1: No self-connections
      if (source === target) return { valid: false, reason: 'Self-connections are not allowed' }

      // Step 2: No null handles
      if (!sourceHandle || !targetHandle) return { valid: false, reason: 'Missing handle' }

      // Step 3: Direction enforcement
      // Convention: output handles start with "output-", input handles start with "input-"
      if (!sourceHandle.startsWith('output-')) return { valid: false, reason: 'Connection must start from an output handle' }
      if (!targetHandle.startsWith('input-')) return { valid: false, reason: 'Connection must end at an input handle' }

      // Step 4: No duplicate connections
      const duplicate = existingEdges.some(
        e => e.source === source && e.target === target
          && e.sourceHandle === sourceHandle && e.targetHandle === targetHandle
      )
      if (duplicate) return { valid: false, reason: 'Duplicate connection' }

      // Step 5: No cycles (DAG)
      if (wouldCreateCycle(source, target, existingEdges)) {
        return { valid: false, reason: 'Connection would create a cycle' }
      }

      return { valid: true }
    }

    export function wouldCreateCycle(
      source: string,
      target: string,
      existingEdges: { source: string; target: string }[]
    ): boolean {
      // BFS from target to see if we can reach source
      // If adding source->target, check if there's already a path from target->source
      const adjacency = new Map<string, string[]>()
      for (const edge of existingEdges) {
        const neighbors = adjacency.get(edge.target) || []
        neighbors.push(edge.source)
        adjacency.set(edge.target, neighbors)
      }
      // Also add the proposed edge in reverse direction for traversal
      const queue = [target]
      const visited = new Set<string>()
      while (queue.length > 0) {
        const current = queue.shift()!
        if (current === source) return true // cycle found
        if (visited.has(current)) continue
        visited.add(current)
        const neighbors = adjacency.get(current) || []
        queue.push(...neighbors)
      }
      return false
    }

    export function isConnectionValid(params: {
      source: string; target: string; sourceHandle: string | null; targetHandle: string | null
    }): boolean {
      return validateConnection(
        params.source, params.target, params.sourceHandle, params.targetHandle, []
      ).valid
    }
    ```

    The cycle detection works by checking if there's already a path from the target node back to the source node in the existing graph. If adding edge A->B, we BFS from B to see if we can reach A. If yes, adding A->B creates a cycle.
