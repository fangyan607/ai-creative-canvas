    2. Add implementations inside the store body (immer `set`/`get`), right after the `setFocusMode` implementation:

    ```typescript
    createGroup: (name, position) => {
      const id = crypto.randomUUID()
      const data: GroupNodeData = { nodeType: 'group', name, collapsed: false, params: [] }
      // Per D-07: group nodes are stored in the same flat nodes array
      const node: NodeGraphNode = { id, type: 'group' as NodeType, position, data, parentId: null }

      set((state) => {
        state.nodes.push(node)
      })

      return id
    },

    addToGroup: (nodeId, groupId) =>
      set((state) => {
        const node = state.nodes.find((n) => n.id === nodeId)
        if (!node) return
        // Set parentId to groupId. React Flow uses this for parent-relative positioning.
        node.parentId = groupId
      }),

    removeFromGroup: (nodeId) =>
      set((state) => {
        const node = state.nodes.find((n) => n.id === nodeId)
        if (!node) return
        node.parentId = null
      }),

    renameGroup: (groupId, name) =>
      set((state) => {
        const node = state.nodes.find((n) => n.id === groupId)
        if (!node || node.type !== 'group') return
        const data = node.data as GroupNodeData
        data.name = name
      }),

    setGroupCollapsed: (groupId, collapsed) =>
      set((state) => {
        const node = state.nodes.find((n) => n.id === groupId)
        if (!node || node.type !== 'group') return
        const data = node.data as GroupNodeData
        data.collapsed = collapsed
      }),
    ```

    **IMPORTANT:** The `removeNode` action must also filter out the node's children when a group node is deleted. Update the existing `removeNode` action to also handle group removal:

    ```typescript
    removeNode: (id) =>
      set((state) => {
        // Find children (nodes with this group's id as parentId)
        const childIds = state.nodes
          .filter((n) => n.parentId === id)
          .map((n) => n.id)

        // Remove the node + its children + their edges
        const toRemove = new Set([id, ...childIds])
        state.nodes = state.nodes.filter((n) => !toRemove.has(n.id))
        state.edges = state.edges.filter(
          (e) => !toRemove.has(e.source) && !toRemove.has(e.target),
        )
      }),
    ```

    4. Update `undo` to restore engine state:
    ```typescript
    undo: () => {
      const state = get()
      if (state.currentIndex <= 0) return

      const newIndex = state.currentIndex - 1
      const snapshot = state.snapshots[newIndex]
      if (snapshot) {
        useCanvasStore.getState().loadSerialized(snapshot.canvas)
        if (snapshot.nodeGraph) {
          useNodeGraphStore.getState().loadSerialized(snapshot.nodeGraph)
        }
        if (snapshot.engine) {  // NEW
          useEngineStore.getState().loadSerialized(snapshot.engine)
        }
        set({ currentIndex: newIndex })
      }
    },
    ```

    5. Update `redo` similarly:
    ```typescript
    redo: () => {
      const state = get()
      if (state.currentIndex >= state.snapshots.length - 1) return

      const newIndex = state.currentIndex + 1
      const snapshot = state.snapshots[newIndex]
      if (snapshot) {
        useCanvasStore.getState().loadSerialized(snapshot.canvas)
        if (snapshot.nodeGraph) {
          useNodeGraphStore.getState().loadSerialized(snapshot.nodeGraph)
        }
        if (snapshot.engine) {  // NEW
          useEngineStore.getState().loadSerialized(snapshot.engine)
        }
        set({ currentIndex: newIndex })
      }
    },
    ```

    **IMPORTANT:** The engine field is optional (`engine?`) in the snapshot interface to maintain backward compatibility with existing snapshots stored in IndexedDB that don't have engine state. Always check `if (snapshot.engine)` before restoring.

    Per D-10: Engine execution events do NOT create history snapshots. Only topology-changing operations do. This is already handled by the topology operations calling `captureSnapshot()` explicitly — the engine itself never calls `captureSnapshot()`.
