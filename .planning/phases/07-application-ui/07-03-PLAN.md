        return db.promptTemplates.orderBy('updatedAt').reverse().toArray()
      },
      async save(template: Omit<PromptTemplateRecord, 'id' | 'createdAt' | 'updatedAt'>): Promise<number> {
        const id = await db.promptTemplates.add({ ...template, createdAt: new Date(), updatedAt: new Date() })

        await db.promptTemplates.delete(id)
      },
    }
